        [y := t]x &= x \\
        [y := t]y &= t \\
        [y := t]\mathfrak{b}(\kappa_1, x : t_1, t_2) &= \mathfrak{b}(\kappa_1, x : [y := t]t_1, [y := t]t_2)  \\
        [y := t]\mathfrak{c}(\kappa_2, t_1, \ldots, t_{\mathfrak{a}(\kappa_2)}) &= \mathfrak{c}(\kappa_2, [y := t]t_1, \ldots, [y := t]t_{\mathfrak{a}(\kappa_2)})
    \end{align*}
