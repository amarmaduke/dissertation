
\Rule{\PseTypeStar}
    {\color{white}{\_}}
    {\star\psetype}
    {}

\Rule{\PseTypeKind}
    {\color{white}{\_}}
    {\kind\psetype}
    {}

\Rule{\PseTypeVar}
    {\color{white}{\_}}
    {x\psetype}
    {}

\Rule{\PseTypeLam}
    {
        \der{\D{1}}{t_1 \psetype} \\
        \der{\D{2}}{t_2 \psetype}
    }
    {\abs{\lambda_\tau}{x}{t_1}{t_2}\psetype}
    {}

\Rule{\PseTypeApp}
    {
        \der{\D{1}}{t_1 \psetype}
    }
    {\app{t_1}{\tau}{t_2}\psetype}
    {}

\Rule{\PseTypePi}
    {
        \der{\D{1}}{t_1 \psetype} \\
        \der{\D{2}}{t_2 \psetype}
    }
    {(x : t_1) \to_m t_2\psetype}
    {}

\Rule{\PseTypeInt}
    {
        \der{\D{1}}{t_1 \psetype} \\
        \der{\D{2}}{t_2 \psetype}
    }
    {(x : t_1) \cap t_2\psetype}
    {}

\Rule{\PseTypeEq}
    {
        \der{\D{1}}{t_2 \psetype}
    }
    {t_1 =_{t_2} t_3\psetype}
    {}
