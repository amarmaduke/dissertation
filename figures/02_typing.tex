
\newcommand{\Rule}[4]{
    \newcommand{#1}[1][a]{
        \if##1a
        \renewcommand{\der}[2]{####2}
        \else
        \renewcommand{\der}[2]{\stackrel{####1}{####2}}
        \fi
        \inferrule{#2}{#3}
        \textsc{#4}
        \renewcommand{\der}[2]{####2}
    }
}

\Rule{\VarRule}
    {\der{\D{1}}{(x : A) \in \Gamma}}
    {\Gamma \vdash x : A}
    {Var}

\Rule{\TermLambdaRule}
    {
        \der{\D{1}}{0\Gamma \vdash (x \type{\sigma} A) \to B \type{} \star}
        \\ \der{\D{2}}{\Gamma, x \type{\sigma} A \vdash t \type{m} B \synth \Gamma, x \type{\sigma \downarrow} A}
    }
    {\Gamma \vdash \absu{\lambda}{x}{t} \type{m} (x \type{\sigma} A) \to B \synth \Gamma}
    {TermLambda}

\newcommand{\TypeLambdaRule}{
    \inferrule
    {
        0\Gamma A \type{\omega} k
        \\ \Gamma, x \type{\omega} A \vdash t : \type{\omega} B
    }
    {
        \Gamma \vdash \absu{\lambda}{x}{t} \type{\omega} (x \type{\omega} A) \synth \Gamma
    }
    \textsc{TypeLambda}
}

\[
    \begin{array}{c}
        \VarRule \\
        \VarRule[*] \\
        $\TermLambdaRule[*]$
    \end{array}
\]
