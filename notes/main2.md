# Type System


    (m x : A) ∈ Γ
    --------------------
    Γ ⊢ x :_m A |> Γ[x]




    Γ ⊢ (∞ x : A) -> B :_∞ □
    (∞ x):A,Γ ⊢ t :_∞ B |> (∞ x):A,Δ
    ---------------------------------
    Γ ⊢ λ (∞ x : A). t :_∞ (∞ x : A) -> B |> Δ


    Γ ⊢ (n x : A) -> B :_∞ *
    (n x):A,Γ ⊢ t :_m B |> (n' x):A,Δ
    ------------------------------------------
    Γ ⊢ λ (n x : A). t :_m (n x : A) -> B |> Δ
